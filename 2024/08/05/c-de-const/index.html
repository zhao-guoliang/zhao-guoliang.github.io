C++ const е…ій”®еӯ—е°Ҹз»“

const жҳҜ constant зҡ„зј©еҶҷпјҢжң¬ж„ҸжҳҜдёҚеҸҳзҡ„гҖҒдёҚжҳ“ж”№еҸҳзҡ„ж„ҸжҖқгҖӮеңЁ C++ дёӯпјҢ

const з”ЁжқҘдҝ®йҘ°еҶ…зҪ®зұ»еһӢеҸҳйҮҸгҖҒиҮӘе®ҡд№үеҜ№иұЎгҖҒжҲҗе‘ҳеҮҪж•°гҖҒиҝ”еӣһеҖјгҖҒеҮҪж•°еҸӮж•°гҖӮ

1. CиҜӯиЁҖзҡ„ const зү№зӮ№

const int a = 10; // дёҚиҰҒжҠҠaзңӢжҲҗеёёйҮҸ
// aзҡ„жң¬иҙЁжҳҜеҸҳйҮҸпјҢеҸӘжҳҜеҸӘиҜ»еҸҳйҮҸ

CиҜӯиЁҖзҡ„ const дҝ®йҘ°е…ЁеұҖеҸҳйҮҸй»ҳи®ӨжҳҜпјҲеӨ–йғЁй“ҫжҺҘзҡ„пјүгҖӮ

fun.c

// CиҜӯиЁҖзҡ„constдҝ®йҘ°е…ЁеұҖеҸҳйҮҸй»ҳи®ӨжҳҜпјҲеӨ–йғЁй“ҫжҺҘзҡ„пјү
// еӨ–йғЁй“ҫжҺҘ: е…¶д»–жәҗж–Үд»¶еҸҜд»ҘдҪҝз”Ё
const int num = 100; // еҸӘиҜ»зҡ„е…ЁеұҖеҸҳйҮҸпјҢеҶ…еӯҳж”ҫеңЁж–Үеӯ—еёёйҮҸеҢәпјҲеҶ…еӯҳз©әй—ҙеҸӘиҜ»пјү

main.c

// еҜ№fun.cдёӯзҡ„numиҝӣиЎҢеЈ°жҳҺ(дёҚиҰҒиөӢеҖј)
extern const int num;
void test03() {
    printf("num = %d\n", num);
    // num = 200; // err numеҸӘиҜ»

    // CиҜӯиЁҖдёӯconstдҝ®йҘ°еҸҳйҮҸеҗҚпјҢиҜҙжҳҺеҸҳйҮҸеҗҚдёәеҸӘиҜ»пјҲз”ЁжҲ·дёҚиғҪйҖҡиҝҮеҸҳйҮҸеҗҚdataиҝӣиЎҢиөӢеҖјпјү
    const int data = 100; // еұҖйғЁеҸӘиҜ»еҸҳйҮҸпјҢеҶ…еӯҳеңЁж ҲеҢәпјҲеҶ…еӯҳеҸҜиҜ»еҸҜеҶҷпјү
    // data = 200; // err

    printf("data = %d\n", data);
    // дҪҶжҳҜпјҡеҰӮжһңзҹҘйҒ“dataзҡ„ең°еқҖпјҢеҸҜд»ҘйҖҡиҝҮең°еқҖй—ҙжҺҘең°дҝ®ж”№dataжүҖеҜ№еә”з©әй—ҙзҡ„еҶ…е®№
    int *p = (int *)&data;
    *p = 2000;
    printf("data = %d\n", data); // ok 2000
}

жҖ»з»“пјҡ

  1. const дҝ®йҘ°е…ЁеұҖеҸҳйҮҸ numпјҢеҸҳйҮҸеҗҚеҸӘиҜ»пјҢеҶ…еӯҳз©әй—ҙеңЁж–Үеӯ—еёёйҮҸеҢәпјҲеҸӘиҜ»пјүпјҢдёҚиғҪйҖҡиҝҮ num зҡ„ең°еқҖдҝ®ж”№з©әй—ҙеҶ…е®№гҖӮ
  2. const дҝ®йҘ°еұҖйғЁеҸҳйҮҸ dataпјҢеҸҳйҮҸеҗҚеҸӘиҜ»пјҢеҶ…еӯҳз©әй—ҙеңЁж ҲеҢәпјҲеҸҜиҜ»еҸҜеҶҷпјүпјҢеҸҜд»ҘйҖҡиҝҮ data ең°еқҖй—ҙжҺҘең°дҝ®ж”№з©әй—ҙеҶ…е®№гҖӮ

2. C++дёӯзҡ„ const ж·ұе…ҘзҗҶи§Ј

fun.cpp

// constдҝ®йҘ°зҡ„е…ЁеұҖеҸҳйҮҸй»ҳи®ӨжҳҜеҶ…йғЁй“ҫжҺҘпјҲеҸӘеңЁеҪ“еүҚжәҗж–Үд»¶жңүж•ҲпјҢдёҚиғҪзӣҙжҺҘз”ЁдәҺе…¶д»–жәҗж–Үд»¶пјү
// const int num = 100;
// еҰӮжһңеҝ…йЎ»з”ЁеңЁе…¶д»–жәҗж–Үд»¶дҪҝз”ЁеҸӘиҜ»зҡ„е…ЁеұҖеҸҳйҮҸпјҢеҝ…йЎ»еҠ  extern е°Ҷ num иҪ¬жҚўжҲҗеӨ–йғЁй“ҫжҺҘ
extern const int num = 100;

main.cpp

// еЈ°жҳҺ
extern const int num;
struct Person {
    int num;
    char name[32];
};

void test04() {
    cout << "е…ЁеұҖnum = " << num << endl; // err дёҚиҜҶеҲ«num

    // 1. C++дёӯпјҢеҜ№дәҺеҹәзЎҖзұ»еһӢпјҢзі»з»ҹдёҚдјҡз»ҷdataејҖиҫҹз©әй—ҙпјҢdataж”ҫеҲ°з¬ҰеҸ·иЎЁдёӯ
    const int data = 10;
    // data = 100; // err еҸӘиҜ»
    cout << "data = " << data << endl;

    // 2. C++дёӯеҪ“еҜ№dataеҸ–ең°еқҖзҡ„ж—¶еҖҷпјҢзі»з»ҹе°ұдјҡз»ҷdataејҖиҫҹз©әй—ҙ
    int *p = (int *)&data;
    *p = 2000;
    cout << "*p = " << *p << endl; // з©әй—ҙеҶ…е®№дҝ®ж”№жҲҗеҠҹ 2000
    cout << "data = " << data << endl; // data иҝҳжҳҜ10пјҢдёәе•Ҙпјҹ

    // 2. еҪ“д»ҘеҸҳйҮҸзҡ„еҪўејҸеҲқе§ӢеҢ–constдҝ®йҘ°зҡ„еҸҳйҮҸпјҢзі»з»ҹдјҡдёәе…¶ејҖиҫҹз©әй—ҙ
    int b = 200;
    const int a = b; // зі»з»ҹзӣҙжҺҘдёәaејҖиҫҹз©әй—ҙпјҢиҖҢдёҚдјҡжҠҠaж”ҫе…Ҙз¬ҰеҸ·иЎЁдёӯ
    p = (int *)&a;
    *p = 3000;
    cout << "*p = " << *p << endl; // 3000
    cout << "a = " << a << endl; // 3000

    // 3. constдҝ®йҘ°иҮӘе®ҡд№үж•°жҚ®зұ»еһӢпјҲз»“жһ„дҪ“гҖҒеҜ№иұЎпјүпјҢзі»з»ҹдјҡеҲҶй…Қз©әй—ҙ
    const Person per = {100, "lucy"};
    // per.num = 1000; // err
    cout << "num = " << per.num << ", name = " << per.name << endl; // 100 lucy
    Person *p1 = (Person *)&per;
    p1->num = 2000;
    cout << "num = " << per.num << ", name = " << per.name << endl; // 2000 lucy
}

иҝҗиЎҢз»“жһңпјҡ

C++зҡ„constжҖ»з»“пјҡ

  1. const int data = 10; // dataе…Ҳж”ҫе…Ҙз¬ҰеҸ·иЎЁ
  2. еҰӮжһңеҜ№dataеҸ–ең°еқҖпјҢзі»з»ҹжүҚдјҡз»ҷdataејҖиҫҹз©әй—ҙ
  3. const int a = b; // bжҳҜеҸҳйҮҸеҗҚпјҢзі»з»ҹзӣҙжҺҘз»ҷaејҖиҫҹз©әй—ҙпјҢиҖҢдёҚж”ҫе…Ҙз¬ҰеҸ·иЎЁ
  4. const дҝ®йҘ°иҮӘе®ҡд№үж•°жҚ®пјҢзі»з»ҹдёәиҮӘе®ҡд№үж•°жҚ®ејҖиҫҹз©әй—ҙ
